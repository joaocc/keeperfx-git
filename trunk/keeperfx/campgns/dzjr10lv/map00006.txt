REM ********** Triggers & End game **********

SET_TIMER(PLAYER1,TIMER0)
SET_TIMER(PLAYER1,TIMER1)
SET_TIMER(PLAYER1,TIMER2)

IF(PLAYER1,TIMER0 >= 450)
    IF(PLAYER1,TOTAL_CREATURES < 28)
        IF(PLAYER1,TOTAL_CREATURES > 0)
            NEXT_COMMAND_REUSABLE
            ADD_CREATURE_TO_LEVEL(PLAYER1,SKELETON,1,1,3,0)
            NEXT_COMMAND_REUSABLE
            SET_TIMER(PLAYER1,TIMER0)
        ENDIF
    ENDIF
ENDIF

IF(PLAYER1,TIMER1 >= 500)
    IF(PLAYER1,TOTAL_CREATURES < 28)
        IF(PLAYER1,TOTAL_CREATURES > 0)
            NEXT_COMMAND_REUSABLE
            ADD_CREATURE_TO_LEVEL(PLAYER1,GHOST,1,1,4,0)
            NEXT_COMMAND_REUSABLE
            SET_TIMER(PLAYER1,TIMER1)
        ENDIF
    ENDIF
ENDIF

IF(PLAYER1,TIMER2 >= 1200)
    IF(PLAYER1,TOTAL_CREATURES < 28)
        IF(PLAYER1,TOTAL_CREATURES > 0)
            NEXT_COMMAND_REUSABLE
            ADD_CREATURE_TO_LEVEL(PLAYER1,GHOST,1,1,6,0)
            NEXT_COMMAND_REUSABLE
            SET_TIMER(PLAYER1,TIMER2)
        ENDIF
    ENDIF
ENDIF

    ADD_TUNNELLER_PARTY_TO_LEVEL(PLAYER_GOOD,ONE,2,DUNGEON_HEART,0,3,300)

IF(PLAYER_GOOD,BATTLES_LOST >= 55)
    ADD_TUNNELLER_PARTY_TO_LEVEL(PLAYER_GOOD,ONE,3,DUNGEON_HEART,0,3,300)
ENDIF

SET_FLAG(PLAYER_GOOD,FLAG0,0)

IF(PLAYER_GOOD,TOTAL_CREATURES == 0)
    ADD_PARTY_TO_LEVEL(PLAYER_GOOD,TWO,-1,2)
    ADD_PARTY_TO_LEVEL(PLAYER_GOOD,THREE,-1,1)
    ADD_PARTY_TO_LEVEL(PLAYER_GOOD,FOUR,-1,1)
    ADD_CREATURE_TO_LEVEL(PLAYER_GOOD,FAIRY,4,3,8,0)
    SET_FLAG(PLAYER_GOOD,FLAG0,1)
ENDIF

IF(PLAYER_GOOD,FLAG0 == 1)
    IF(PLAYER_GOOD,AVATAR == 0)
        WIN_GAME
    ENDIF
ENDIF
